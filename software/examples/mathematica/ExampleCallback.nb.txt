host="localhost"

uid="XYZ"(*Change XYZ to the UID of your LED Strip Bricklet 2.0*)

(*Create IPConnection and device object*)
ipcon=NETNew["Tinkerforge.IPConnection"]
ls=NETNew["Tinkerforge.BrickletLEDStripV2",uid,ipcon]
ipcon@Connect[host,port]

(*Set frame duration to 50ms (20 frames per second)*)
ls@SetFrameDuration[50]

(*Use frame rendered callback to move the active LED every frame*)
FrameRenderedCB[sender_,length_]:=
 Print["Length: "<>ToString[length]]
AddEventHandler[ls@FrameRenderedCallback,FrameRenderedCB]

Input["Click OK to exit"]

(*Clean up*)
ipcon@Disconnect[]
ReleaseNETObject[ls]
ReleaseNETObject[ipcon]
